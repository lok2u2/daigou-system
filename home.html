async function loadOrders(){ const keyword = document.querySelector(".search").value.toLowerCase() const res = await fetch(API + "/api/order/list") const data = await res.json() const box = document.getElementById("orders") box.innerHTML = "" // 模糊搜索 const filtered = data.filter(o => { return ( (o.customer || "").toLowerCase().includes(keyword) || (o.product || "").toLowerCase().includes(keyword) || (o.platform_order_id || "").toLowerCase().includes(keyword) ) }) filtered.forEach(o => { let color = "#666" if(o.status === "pending"){ color = "orange" } if(o.status === "arrived"){ color = "blue" } if(o.status === "notified"){ color = "purple" } if(o.status === "shipped"){ color = "green" } box.innerHTML += `
${o.customer || "-"}
${o.product || "-"}
平台：${o.platform || "-"}
订单号： ${o.platform_order_id || "-"}
运单尾号： ${o.tracking_last4 || "-"}
微信/Zalo： ${o.phone || "-"}
人民币： ¥${o.amount_cny || 0}
越南盾： ${o.amount_vnd || 0}₫
状态：${o.status}

` }) }
